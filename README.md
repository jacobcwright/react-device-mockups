## Install

```bash
npm install --save react-device-mockups
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-device-mockups'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
